    while (z->prox!=NULL){
        z=z->prox;
    }
    LInt bottom = z;
    LInt low = bottom;
    z=top;
    while (low!=top){
        while (z->prox!=low){
        z=z->prox;
    }
    low->prox = z;
    low=low->prox;
    z=top;
    }
    top->prox = NULL;

    return bottom;
}

        tester = depth(a->dir,x);
        if (tester > -1)
            level = 1 + depth(a->dir, x);
        else
            level =-1;
    }
    else if (x < a->valor){
        tester = depth(a->esq,x);

        else
            level =-1;
    }

}

LInt nivel (ABin, int)

LInt nivel (ABin a, int n){
    LInt l;
    if (!a)
        return NULL;
    if (n==1){
        l = newLInt(a->valor, NULL);
        return l;
    }
    LInt head;
    l = nivel(a->esq, n-1);
    if (l){
        head = l;
        while (l->prox)
            l = l->prox;
        l->prox = nivel(a->dir, n-1);
    }
    else {
        l = nivel(a->dir, n-1);
        head = l;
    }
    return head;
}
