LInt reverse (LInt z){
    if (z==NULL || z->prox==NULL)
        return z;
    LInt top = z;
    while (z->prox!=NULL){
        z=z->prox;
    }
    LInt bottom = z;
    LInt low = bottom;
    z=top;
    while (low!=top){
        while (z->prox!=low){
        z=z->prox;
    }
    low->prox = z;
    low=low->prox;
    z=top;
    }
    top->prox = NULL;

    return bottom;
}

LInt merge(LInt a, LInt b){
    if (a==NULL)
        return b;
    if (b==NULL)
        return a;
    LInt head;
    if (a->valor < b->valor){
        head = a;
        a = a->prox;
    }
    else{
        head = b;
        b = b->prox;
    }
    LInt fixed_head = head;
    while (a!=NULL && b!=NULL){
        if (a->valor < b->valor){
            head->prox = a;
            head = head->prox;
            a = a->prox;
        }
        else if (a->valor > b->valor){
            head->prox = b;
            head = head->prox;
            b = b->prox;
        }
    }
    if (a == NULL)
        head->prox = b;
    else if (b==NULL)
        head->prox = a;
    return fixed_head;
}

        return 0;

    ll = ll->prox;

                break;

        free(ll);
        ll = ante->prox;
        }
    }
    return count;
}
